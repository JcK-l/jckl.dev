export type HeroFaceImageLayout = {
  x: number;
  y: number;
  width: number;
  assetWidth: number;
  assetHeight: number;
  src: string;
  alt: string;
  srcSet?: string;
  sizes?: string;
};
 
type HeroFaceMaskLayerProps = {
  id: string;
  className: string;
  fill: string;
  path: string;
  offsetX: number;
  offsetY: number;
};
 
type HeroFaceFrameProps = {
  imageLayout: HeroFaceImageLayout;
  hideImage: boolean;
};
 
const getHeroImageStyle = (layout: HeroFaceImageLayout) => {
  return {
    aspectRatio: `${layout.assetWidth} / ${layout.assetHeight}`,
    left: `${(layout.x / HERO_VIEWBOX_WIDTH) * 100}%`,
    top: `${(layout.y / HERO_VIEWBOX_HEIGHT) * 100}%`,
    width: `${(layout.width / HERO_VIEWBOX_WIDTH) * 100}%`,
  } as const;
};
 
const HeroFaceMaskLayer = ({
  id,
  className,
  fill,
  path,
  offsetX,
  offsetY,
}: HeroFaceMaskLayerProps) => {
  const hasOffset = offsetX !== 0 || offsetY !== 0;
 
  return (
    <svg
      className={`pointer-events-none absolute inset-0 ${className}`}
      viewBox={`0 0 ${HERO_VIEWBOX_WIDTH} ${HERO_VIEWBOX_HEIGHT}`}
      version="1.1"
      xmlns="http://www.w3.org/2000/svg"
      aria-hidden="true"
    >
      <defs>
        <mask
          id={id}
          x="0"
          y="0"
          width={HERO_VIEWBOX_WIDTH}
          height={HERO_VIEWBOX_HEIGHT}
          maskUnits="userSpaceOnUse"
          maskContentUnits="userSpaceOnUse"
        >
          <rect
            x="0"
            y="0"
            width={HERO_VIEWBOX_WIDTH}
            height={HERO_VIEWBOX_HEIGHT}
            fill="white"
          />
          <g transform={hasOffset ? `translate(${offsetX} ${offsetY})` : undefined}>
            <path d={path} fill="black" />
          </g>
        </mask>
      </defs>
      <rect
        x="0"
        y="0"
        width={HERO_VIEWBOX_WIDTH}
        height={HERO_VIEWBOX_HEIGHT}
        mask={`url(#${id})`}
        style={{
          filter: "var(--hero-face-shadow-filter)",
          fill,
        }}
      />
    </svg>
  );
};
 
export const HeroFaceFrame = ({
  imageLayout,
  hideImage,
}: HeroFaceFrameProps) => {
  const maskPrefix = useId().replace(/:/g, "");
  const ref = useRef<HTMLDivElement>(null);
  const { scrollYProgress } = useScroll({
    target: ref,
    offset: ["0% 0%", "100% 0%"],
  });
  const imageY = useTransform(scrollYProgress, [0, 1], ["0vh", "10vh"]);
 
  return (
    <div
      ref={ref}
      className="relative z-30 mb-8 h-auto w-10/12 select-none overflow-hidden bg-bgColor sm:w-9/12 md:mb-10 md:w-8/12 lg:w-6/12 2xl:w-5/12 desk:w-4/12 desk-l:w-4/12"
    >
      <svg
        viewBox={`0 0 ${HERO_VIEWBOX_WIDTH} ${HERO_VIEWBOX_HEIGHT}`}
        version="1.1"
        xmlns="http://www.w3.org/2000/svg"
        className="relative -z-30"
        aria-hidden="true"
      >
        <path style={{ fill: "none" }} d="M 0,0 V 743 H 900 V 0 Z" />
      </svg>
 
      <motion.div
        style={{ y: imageY }}
        className={`absolute inset-0 mix-blend-screen ${
          hideImage ? "pointer-events-none invisible" : ""
        }`}
      >
        <img
          src={imageLayout.src}
          srcSet={imageLayout.srcSet}
          sizes={imageLayout.sizes}
          alt={imageLayout.alt}
          className="absolute block h-auto max-w-none"
          decoding="async"
          fetchpriority="high"
          loading="eager"
          style={getHeroImageStyle(imageLayout)}
        />
      </motion.div>
 
      {heroMaskLayers.map((layer) => {
        return (
          <HeroFaceMaskLayer
            key={layer.key}
            id={`${maskPrefix}-${layer.key}`}
            className={layer.className}
            fill={layer.fill}
            path={layer.path}
            offsetX={layer.offsetX}
            offsetY={layer.offsetY}
          />
        );
      })}
    </div>
  );
};
